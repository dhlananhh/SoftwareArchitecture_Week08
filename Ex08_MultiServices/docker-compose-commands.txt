# Khởi chạy stack: Build image Node.js và khởi chạy các containers
    docker compose up -d --build

# Kiểm tra: Xem log của Node.js app để thấy thông báo kết nối DB:
    docker compose logs nodejs_connector_app -f
    Bạn nên thấy log "Connected to MySQL database successfully!" và server bắt đầu lắng nghe.
        Mở trình duyệt và truy cập http://localhost:3001.
        Bạn sẽ thấy thông báo trạng thái kết nối DB

# Clean up:
    docker compose down -v
